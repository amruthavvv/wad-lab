CSS /* General page styling */ body { font-family: 'Segoe UI', Arial, sans-serif; background-color: lightgray; color: black; margin: 0; padding: 0; } /* Header styling */ header { background-color: dodgerblue; color: white; text-align: center; padding: 20px; } /* Box styling example */ .intro-box { background-color: white; border: 2px solid dodgerblue; border-radius: 8px; margin: 20px; padding: 15px; } /* Styled list */ .list-section { margin: 20px; padding: 10px; background-color: lightblue; border: 1px solid skyblue; border-radius: 6px; } .styled-list { list-style-type: square; padding-left: 25px; } .styled-list li { background-color: lightskyblue; margin: 8px 0; padding: 8px; border-radius: 4px; transition: 0.3s; } .styled-list li:hover { background-color: deepskyblue; cursor: pointer; } /* Styled form */ .form-section { background-color: moccasin; border: 1px solid orange; border-radius: 8px; margin: 20px; padding: 15px; } form label { display: inline-block; width: 100px; font-weight: bold; } input[type="text"], input[type="email"], select { padding: 6px; border: 1px solid gray; border-radius: 4px; } input[type="submit"] { background-color: dodgerblue; color: white; border: none; padding: 8px 16px; border-radius: 4px; cursor: pointer; } input[type="submit"]:hover { background-color: royalblue; } /* Table styling */ .table-section { margin: 20px; padding: 10px; background-color: honeydew; border: 1px solid yellowgreen; border-radius: 6px; } table { width: 100%; border-collapse: collapse; } th, td { border: 1px solid yellowgreen; padding: 10px; text-align: center; } th { background-color: yellowgreen; } tr:nth-child(even) { background-color: khaki; } /* Footer styling */ footer { background-color: black; color: white; text-align: center; padding: 10px; margin-top: 20px; }
